from __future__ import annotations
import json, os, uuid, subprocess, tempfile
from pathlib import Path
import google.generativeai as genai
from googleapiclient.discovery import build


_ER16_PROMPT = """You are an embodied-reasoning assistant.
Given a task description, return ONLY valid JSON with keys:
task_goal (str), affordances (list[str]), success_criteria (str),
search_queries (list[str], top 3 YouTube queries to find reference videos).
Task: {prompt}"""


class IngestService:
    def __init__(
        self,
        gemini_api_key: str,
        youtube_api_key: str,
        supabase_url: str,
        supabase_key: str,
    ) -> None:
        genai.configure(api_key=gemini_api_key)
        self._gemini_model = genai.GenerativeModel("gemini-robotics-er-1.6")
        self._youtube_api_key = youtube_api_key

    def analyze_prompt(self, prompt: str) -> dict:
        resp = self._gemini_model.generate_content(
            _ER16_PROMPT.format(prompt=prompt)
        )
        return json.loads(resp.text)

    def search_youtube(self, query: str) -> str:
        yt = build("youtube", "v3", developerKey=self._youtube_api_key)
        result = yt.search().list(
            q=query, part="id,snippet", type="video",
            maxResults=5, videoDuration="short",
        ).execute()
        return result["items"][0]["id"]["videoId"]

    def download_clip(self, video_id: str, dest_dir: Path) -> Path:
        url = f"https://www.youtube.com/watch?v={video_id}"
        out = dest_dir / f"{video_id}.mp4"
        subprocess.run(
            ["yt-dlp", "-f", "mp4", "-o", str(out), url],
            check=True,
        )
        return out

    def run_gvhmr(self, video_url: str) -> str:
        """Dispatch to existing Modal GVHMR endpoint. Returns job_id."""
        from scripts.gvhmr_modal_probe import process_video  # type: ignore
        result = process_video.remote(video_url=video_url)
        return result.get("run_id", str(uuid.uuid4()))
```

from __future__ import annotations
import uuid
from supabase import Client


class EvolutionService:
    def __init__(self, supa: Client) -> None:
        self.supa = supa

    def create(self, run_id: str) -> str:
        evo_id = str(uuid.uuid4())
        resp = (
            self.supa.table("evolutions")
            .insert({"id": evo_id, "run_id": run_id, "status": "pending"})
            .execute()
        )
        return resp.data[0]["id"]

    def get(self, evo_id: str) -> dict:
        resp = (
            self.supa.table("evolutions")
            .select("*")
            .eq("id", evo_id)
            .single()
            .execute()
        )
        return resp.data

    def update_status(self, evo_id: str, status: str) -> None:
        self.supa.table("evolutions").update({"status": status}).eq("id", evo_id).execute()

    def set_best(self, evo_id: str, iter_id: str) -> None:
        self.supa.table("evolutions").update({"best_iteration_id": iter_id}).eq("id", evo_id).execute()

    def add_cost(self, evo_id: str, delta_usd: float) -> None:
        evo = self.get(evo_id)
        new_cost = float(evo.get("total_cost_usd") or 0) + delta_usd
        self.supa.table("evolutions").update({"total_cost_usd": new_cost}).eq("id", evo_id).execute()

    def record_iteration(self, evo_id: str, iter_num: int, result: dict) -> str:
        import uuid as _uuid
        iter_id = str(_uuid.uuid4())
        self.supa.table("iterations").insert({"id": iter_id, "evolution_id": evo_id, "iter_num": iter_num, **result}).execute()
        return iter_id
```

from __future__ import annotations
import subprocess
from pathlib import Path


class CLIOrchestrator:
    def __init__(self, workdir: Path) -> None:
        self.workdir = Path(workdir)

    def edit_files(
        self,
        prompt: str,
        editable: list[str],
        timeout_s: int = 120,
    ) -> dict:
        """Invoke Codex CLI to edit files. Falls back to Claude CLI on failure."""
        file_list = " ".join(editable)
        full_prompt = f"{prompt}\n\nYou may only edit these files: {file_list}"

        # Primary: Codex CLI
        result = subprocess.run(
            ["codex", "exec", full_prompt],
            cwd=self.workdir,
            capture_output=True,
            text=True,
            timeout=timeout_s,
        )
        if result.returncode == 0:
            return {"generator": "codex", "stdout": result.stdout}

        # Fallback: Claude Code CLI
        result = subprocess.run(
            ["claude", "-p", full_prompt, "--allowedTools", "Edit,Read"],
            cwd=self.workdir,
            capture_output=True,
            text=True,
            timeout=timeout_s,
        )
        if result.returncode != 0:
            raise RuntimeError(
                f"Both Codex and Claude CLI failed.\n"
                f"Codex stderr: {result.stderr}\n"
                f"Claude stderr: {result.stderr}"
            )
        return {"generator": "claude-code", "stdout": result.stdout}

    def draft_program_md(self, er16_plan: dict, timeout_s: int = 60) -> tuple[str, str]:
        """Draft program.md from ER 1.6 plan. Returns (content, generator)."""
        import json
        prompt = (
            "You are setting up an autonomous robot design research loop.\n"
            "Draft a program.md file (research agenda) for this task:\n"
            f"{json.dumps(er16_plan, indent=2)}\n\n"
            "The program.md should describe:\n"
            "1. What morphology features to explore\n"
            "2. What controller changes to try\n"
            "3. How to measure progress\n"
            "4. What to avoid (known failure modes)\n"
            "Keep it under 300 words. Plain English."
        )
        result = subprocess.run(
            ["codex", "exec", prompt],
            cwd=self.workdir,
            capture_output=True, text=True, timeout=timeout_s,
        )
        if result.returncode == 0:
            return result.stdout.strip(), "codex"

        result = subprocess.run(
            ["claude", "-p", prompt],
            cwd=self.workdir,
            capture_output=True, text=True, timeout=timeout_s,
        )
        return result.stdout.strip(), "claude-code"
```

from __future__ import annotations
import os, uuid
from pathlib import Path
from fastapi import APIRouter, HTTPException, BackgroundTasks
from pydantic import BaseModel
from demo.supabase_client import supa
from demo.services.evolution_service import EvolutionService
from demo.services.orchestrator import CLIOrchestrator
from demo.services.modal_dispatch import ModalDispatch
from packages.pipeline.types import EvolutionConfig

router = APIRouter(prefix="/evolutions", tags=["evolutions"])
_evo_svc = EvolutionService(supa=supa)
_dispatch = ModalDispatch()

ARTIFACTS_DIR = Path("data/artifacts/evolutions")


class CreateEvolutionRequest(BaseModel):
    run_id: str
    ingest_job_id: str


class ApproveProgramRequest(BaseModel):
    content: str   # final (possibly user-edited) program.md content


@router.post("", status_code=201)
def create_evolution(req: CreateEvolutionRequest) -> dict:
    evo_id = _evo_svc.create(run_id=req.run_id)
    workdir = ARTIFACTS_DIR / evo_id
    workdir.mkdir(parents=True, exist_ok=True)
    # Copy template files
    template = ARTIFACTS_DIR / "template"
    for f in ["prepare.py", "train.py", "morphology_factory.py"]:
        (workdir / f).write_text((template / f).read_text())
    orch = CLIOrchestrator(workdir=workdir)
    # Get ER 1.6 plan from ingest job
    ingest = supa.table("ingest_jobs").select("er16_plan_json").eq("id", req.ingest_job_id).single().execute()
    import json
    plan = json.loads(ingest.data["er16_plan_json"].replace("'", '"'))
    draft, generator = orch.draft_program_md(er16_plan=plan)
    draft_id = str(uuid.uuid4())
    supa.table("program_md_drafts").insert({
        "id": draft_id, "evolution_id": evo_id,
        "generator": generator, "draft_content": draft, "approved": False,
    }).execute()
    return {"evolution_id": evo_id, "draft_id": draft_id, "draft_content": draft}


@router.post("/{evo_id}/approve-program")
def approve_program(evo_id: str, req: ApproveProgramRequest, bg: BackgroundTasks) -> dict:
    workdir = ARTIFACTS_DIR / evo_id
    (workdir / "program.md").write_text(req.content)
    supa.table("program_md_drafts").update(
        {"approved": True, "user_edited_content": req.content}
    ).eq("evolution_id", evo_id).execute()
    _evo_svc.update_status(evo_id, "running")
    bg.add_task(_run_evolution_loop, evo_id)
    return {"status": "running"}


@router.post("/{evo_id}/stop")
def stop_evolution(evo_id: str) -> dict:
    _evo_svc.update_status(evo_id, "stopped")
    return {"status": "stopped"}


@router.post("/{evo_id}/mark-best/{iter_id}")
def mark_best(evo_id: str, iter_id: str) -> dict:
    _evo_svc.set_best(evo_id, iter_id)
    return {"best_iteration_id": iter_id}


@router.get("/{evo_id}")
def get_evolution(evo_id: str) -> dict:
    return _evo_svc.get(evo_id)


def _run_evolution_loop(evo_id: str) -> None:
    """Background task: orchestrator loop."""
    import time, json
    from packages.pipeline.types import EvolutionConfig
    cfg = EvolutionConfig()
    workdir = ARTIFACTS_DIR / evo_id
    orch = CLIOrchestrator(workdir=workdir)
    # Get SMPL URL from ingest job via evolution → run → ingest_job
    evo = _evo_svc.get(evo_id)
    run = supa.table("runs").select("clip_id").eq("id", evo["run_id"]).single().execute()
    clip = supa.table("clips").select("smpl_path").eq("id", run.data["clip_id"]).single().execute()
    smpl_url = clip.data["smpl_path"]

    best_score = -1.0
    no_improve = 0
    t0 = time.time()

    for i in range(cfg.max_iters):
        evo_fresh = _evo_svc.get(evo_id)
        if evo_fresh["status"] in ("stopped", "done"):
            break
        elapsed_h = (time.time() - t0) / 3600
        if elapsed_h >= cfg.max_hours:
            break

        train_py = (workdir / "train.py").read_text()
        morph_py = (workdir / "morphology_factory.py").read_text()
        prev_train = train_py
        prev_morph = morph_py

        # Agent edits files
        context = f"program.md:\n{(workdir / 'program.md').read_text()}\n"
        orch.edit_files(
            prompt=context + f"Iteration {i}. Improve train.py and/or morphology_factory.py.",
            editable=["train.py", "morphology_factory.py"],
        )

        # Dispatch Modal trial
        try:
            result = _dispatch.run_trial(
                evolution_id=evo_id, iter_num=i,
                train_py_source=(workdir / "train.py").read_text(),
                morph_factory_source=(workdir / "morphology_factory.py").read_text(),
                smpl_trajectory_url=smpl_url,
            )
        except Exception as exc:
            # Failed trial: score 0, continue
            supa.table("iterations").insert({
                "id": str(uuid.uuid4()), "evolution_id": evo_id, "iter_num": i,
                "fitness_score": 0.0, "reasoning_log": f"Trial failed: {exc}",
            }).execute()
            continue

        # Persist iteration (Realtime pushes to frontend automatically)
        import difflib
        train_diff = "".join(difflib.unified_diff(
            prev_train.splitlines(keepends=True),
            (workdir / "train.py").read_text().splitlines(keepends=True),
        ))
        morph_diff = "".join(difflib.unified_diff(
            prev_morph.splitlines(keepends=True),
            (workdir / "morphology_factory.py").read_text().splitlines(keepends=True),
        ))
        iter_id = _evo_svc.record_iteration(evo_id, i, {
            "fitness_score": result.fitness_score,
            "tracking_error": result.tracking_error,
            "er16_success_prob": result.er16_success_prob,
            "replay_mp4_url": result.replay_mp4_url,
            "controller_ckpt_url": result.controller_ckpt_url,
            "trajectory_npz_url": result.trajectory_npz_url,
            "reasoning_log": result.reasoning_md,
            "train_py_diff": train_diff,
            "morph_factory_diff": morph_diff,
        })

        # Keep-best
        if result.fitness_score > best_score * (1 + cfg.keep_best_threshold):
            best_score = result.fitness_score
            _evo_svc.set_best(evo_id, iter_id)
            no_improve = 0
        else:
            no_improve += 1

        if no_improve >= 5:
            break

    _evo_svc.update_status(evo_id, "done")
```
